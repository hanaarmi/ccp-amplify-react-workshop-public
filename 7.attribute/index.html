  1. src/component/ccp/ccp.component.jsx 전체 소스코드는 아래와 같습니다.
import React, { Component } from 'react'
import { Auth } from 'aws-amplify'
import 'amazon-connect-streams'
import { ccpAuthLogin } from './ccp.util'

import './ccp.component.scss'

class CCP extends Component {
	constructor(props) {
		super(props)
		this.ccp = React.createRef()
		this.loginFrm = React.createRef()
		this.state = {
			credentials: '',
		}
		this.logMsgs = React.createRef()
	}

	async componentDidMount() {
		const { attributes } = await Auth.currentAuthenticatedUser()
		console.log(attributes.email)
		ccpAuthLogin(attributes.email).then((result) => {
			console.log(result.ccpResult)
			this.setState({ credentials: JSON.stringify(result.ccpResult) })
			this.loginFrm.current.submit()

			window.connect.core.initCCP(this.ccp.current, {
			 	ccpUrl: 'https://[인스턴스URL]/ccp-v2',
			 	loginPopup: false,
			 	region: 'us-east-1',
			 	softphone: {
			 		allowFramedSoftphone: true,
					disableRingtone: false,
				},
			})

			window.connect.contact(function (contact) {
				contact.onConnected(function () {
					document.getElementById(
						'logMsgs'
					).innerHTML = `onConnected(${contact.getContactId()})<br>`
					var attributeMap = contact.getAttributes()
					var note = JSON.stringify(attributeMap['note']['value'])
					var phonenumber = JSON.stringify(
						attributeMap['phonenumber']['value']
					)
					var firstname = JSON.stringify(
						attributeMap['firstname']['value']
					)
					var lastname = JSON.stringify(
						attributeMap['lastname']['value']
					)
					console.log(attributeMap)
					console.log(note)
					document.getElementById(
						'logMsgs'
					).innerHTML += `${note}<br>${phonenumber}<br>${firstname}<br>${lastname}<br>`
				})
			})
		})
	}

	componentWillUnmount() {
		window.connect.core.terminate()
		var iframe = this.ccp.current.firstElementChild
		if (iframe) {
			this.ccp.current.removeChild(iframe)
		}
	}

	render() {
		return (
			<div className="ccp-container">
				<div id="ccp" className="ccp" ref={this.ccp} />
				<div
					id="hidden_iframe_div"
					style={{ visibility: 'hidden' }}
					className="hidden_iframe_div"
				>
					<iframe
						title="hiddenframe"
						src=""
						id="hidden_iframe"
						name="hidden_iframe"
						frameBorder="0"
						style={{ visibility: 'hidden' }}
					></iframe>
				</div>
				<div
					id="loginFrm_div"
					className="loginFrm_div"
					style={{ visibility: 'hidden', textAlign: 'left' }}
				>
					<form
						ref={this.loginFrm}
						id="loginFrm"
						method="POST"
						target="hidden_iframe"
						action={
							'https://[인스턴스URL]/auth/sign-in'
						}
					>
						<label htmlFor="credentials">Credentials:</label>
						<input
							type="text"
							name="credentials"
							id="credentials"
							value={this.state.credentials}
							onChange={(e) => {
								this.setState({ credentials: e.target.value })
							}}
						/>
					</form>
				</div>
				<div id="logMsgsContainer" className="logMsgsContainer">
					<h2>Log Messages</h2>
					<div id="logMsgs" ref={this.logMsgs}></div>
				</div>
			</div>
		)
	}
}

export default CCP
